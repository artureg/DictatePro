001    package com.wiseapps.davacon.ui.adapters;
002    
003    import android.content.Context;
004    import android.view.LayoutInflater;
005    import android.view.View;
006    import android.view.ViewGroup;
007    import android.widget.ArrayAdapter;
008    import android.widget.TextView;
009    import com.wiseapps.davacon.R;
010    import com.wiseapps.davacon.core.wav.WAVFile;
011    
012    import java.util.List;
013    
014    /**
015     * @author varya.bzhezinskaya@gmail.com
016     *         Date: 3/20/14
017     *         Time: 9:28 AM
018     *
019     * TODO generalize
020     */
021    public class TrackAdapter extends ArrayAdapter<WAVFile> {
022    
023        private final TrackActionsListener listener;
024        private final LayoutInflater inflater;
025    
026        public TrackAdapter(Context context, List<WAVFile> tracks, TrackActionsListener listener) {
027            super(context, R.layout.track, tracks);
028    
029            this.listener = listener;
030            this.inflater = (LayoutInflater) context.
031                    getSystemService(Context.LAYOUT_INFLATER_SERVICE);
032        }
033    
034        @Override
035        public View getView(final int position, View convertView, ViewGroup parent) {
036            if (convertView == null) {
037                convertView = inflater.inflate(R.layout.track, null);
038            }
039    
040            WAVFile wav = getItem(position);
041    
042            ((TextView) convertView.findViewById(R.id.track)).
043                    setText(wav.getFile().getName());
044    
045            convertView.setOnClickListener(new View.OnClickListener() {
046                @Override
047                public void onClick(View view) {
048                    if (listener != null) {
049                        listener.onDetails(position);
050                    }
051                }
052            });
053    
054            return convertView;
055        }
056    
057        public interface TrackActionsListener {
058            void onDetails(int position);
059            void onDelete(int position);
060        }
061    }

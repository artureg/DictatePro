001    package com.wiseapps.davacon.core.wav;
002    
003    import com.wiseapps.davacon.logging.LoggerFactory;
004    
005    import java.io.IOException;
006    import java.io.RandomAccessFile;
007    
008    /**
009     * Class to write .wav file.
010     *
011     * @author varya.bzhezinskaya@gmail.com
012     *         Date: 3/30/14
013     *         Time: 11:12 AM
014     */
015    public class WAVFileWriter {
016        private static final String TAG = WAVFileWriter.class.getSimpleName();
017    
018        private RandomAccessFile writer;
019    
020        private transient boolean prepared;
021        private transient boolean consumed;
022    
023        private int totalNumberOfBytes;
024    
025        private final WAVFile wav;
026    
027        public WAVFileWriter(WAVFile wav) {
028            this.wav = wav;
029        }
030    
031        /**
032         * Method to write data to .wav file.
033         *
034         * @param data array of bytes to write to file
035         * @throws IOException
036         */
037        public void write(byte[] data) throws IOException {
038            if (consumed) {
039                throw new IllegalStateException("File writer already consumed!");
040            }
041    
042            if (!prepared) {
043                prepare();
044            }
045    
046            writer.write(data);
047            totalNumberOfBytes += data.length;
048    
049            LoggerFactory.obtainLogger(TAG).
050                    d("write# " + String.format("Total number of bytes written is %d", totalNumberOfBytes));
051        }
052    
053        public void consume() throws IOException {
054            writer.seek(4); // Write size to RIFF header (ChunkSize)
055            writer.writeInt(Integer.reverseBytes(36 + totalNumberOfBytes));
056    
057            writer.seek(40); // Write size to Subchunk2Size field
058            writer.writeInt(Integer.reverseBytes(totalNumberOfBytes));
059    
060            writer.close();
061    
062            consumed = true;
063    
064            LoggerFactory.obtainLogger(TAG).
065                    d("consume# " + String.format("File size is %d", wav.getFile().length()));
066        }
067    
068        private void prepare() throws IOException {
069            writer = new RandomAccessFile(wav.getFile(), "rw");
070            writer.setLength(0);
071    
072            // Set RIFF-header section
073            writer.writeBytes(wav.getChunkID());
074            LoggerFactory.obtainLogger(TAG).
075                    d(String.format("prepare# ChunkID = %s", wav.getChunkID()));
076    
077            writer.writeInt(Integer.reverseBytes(0));
078    
079            writer.writeBytes(wav.getFormat());
080            LoggerFactory.obtainLogger(TAG).
081                    d(String.format("prepare# Format = %s", wav.getFormat()));
082    
083            // Set fmt-subchunk
084            writer.writeBytes(wav.getSubchunk1ID());
085            LoggerFactory.obtainLogger(TAG).
086                    d(String.format("prepare# Subchunk1ID = %s", wav.getSubchunk1ID()));
087    
088            writer.writeInt(Integer.reverseBytes(wav.getSubchunk1Size()));
089            LoggerFactory.obtainLogger(TAG).
090                    d(String.format("prepare# Subchunk1Size = %s", wav.getSubchunk1Size()));
091    
092            writer.writeShort(Short.reverseBytes(wav.getAudioFormat()));
093            LoggerFactory.obtainLogger(TAG).
094                    d(String.format("prepare# AudioFormat = %s", wav.getAudioFormat()));
095    
096            writer.writeShort(Short.reverseBytes(wav.getNumChannels()));
097            LoggerFactory.obtainLogger(TAG).
098                    d(String.format("prepare# NumChannels = %s", wav.getNumChannels()));
099    
100            writer.writeInt(Integer.reverseBytes(wav.getSampleRate()));
101            LoggerFactory.obtainLogger(TAG).
102                    d(String.format("prepare# SampleRate = %s", wav.getSampleRate()));
103    
104            writer.writeInt(Integer.reverseBytes(wav.getByteRate()));
105            LoggerFactory.obtainLogger(TAG).
106                    d(String.format("prepare# ByteRate = %s", wav.getByteRate()));
107    
108            writer.writeShort(Short.reverseBytes(wav.getBlockAlign()));
109            LoggerFactory.obtainLogger(TAG).
110                    d(String.format("prepare# BlockAlign = %s", wav.getBlockAlign()));
111    
112            writer.writeShort(Short.reverseBytes(wav.getBitsPerSample()));
113            LoggerFactory.obtainLogger(TAG).
114                    d(String.format("prepare# BitsPerSample = %s", wav.getBitsPerSample()));
115    
116            // Set data-subchunk
117            writer.writeBytes(wav.getSubchunk2ID());
118            LoggerFactory.obtainLogger(TAG).
119                    d(String.format("prepare# Subchunk2ID = %s", wav.getSubchunk2ID()));
120    
121            writer.writeInt(Integer.reverseBytes(0));
122    
123            prepared = true;
124    
125            LoggerFactory.obtainLogger(TAG).d("prepare# writer.length = " + writer.length());
126        }
127    }

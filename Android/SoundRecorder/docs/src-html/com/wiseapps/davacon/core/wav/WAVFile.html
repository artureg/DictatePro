022    public class WAVFile extends SoundFile {
023        private static final String TAG = WAVFile.class.getSimpleName();
024    
025        public static final int RECORDER_AUDIO_SOURCE = MediaRecorder.AudioSource.MIC;
026        public static final int RECORDER_SAMPLE_RATE_IN_HZ = 44100;
027        public static final int RECORDER_CHANNEL_CONFIG = AudioFormat.CHANNEL_IN_STEREO;
028        public static final int RECORDER_AUDIO_FORMAT = AudioFormat.ENCODING_PCM_16BIT;
029        public static final int RECORDER_BUFFER_SIZE_IN_BYTES =
030                AudioRecord.getMinBufferSize(RECORDER_SAMPLE_RATE_IN_HZ, RECORDER_CHANNEL_CONFIG, RECORDER_AUDIO_FORMAT);
031    
032        private final WAVFileWriter writer;
033        private final WAVFileReader reader;
034    
035        private short numChannels;
036    
037        // TODO set 16
038        private int sampleRate;
039    
040        // 8 bits = 8, 16 bits = 16, etc.
041        // TODO we should set it somehow probably, TODO try with 8
042        private final short mBitsPerSample = 16;
043    
044        private int subchunk2Size;
045    
046        private transient byte[] data;
047    
048        private WAVFile(File file, int audioFormat, int channelConfig, int sampleRateInHz) throws IOException {
049            if (audioFormat != AudioFormat.ENCODING_PCM_16BIT) {
050                throw new IllegalArgumentException("Incorrect audio format, working with PCM 16 bit only!");
051            }
052    
053            numChannels = (short) (channelConfig == AudioFormat.CHANNEL_IN_STEREO ? 2 : 1);
054            sampleRate = sampleRateInHz;
055    
056            writer = new WAVFileWriter(this);
057            reader = new WAVFileReader(this);
058    
059            setFile(file);
060        }
061    
062        /**
063         * Method to read .wav file.
064         */
065        public void read() {
066            reader.read();
067        }
068    
069        /**
070         * Method to write data to .wav file.
071         *
072         * @param data array of bytes to write to file
073         * @throws IOException
074         */
075        public void write(byte[] data) throws IOException {
076            writer.write(data);
077        }
078    
079        /**
080         * Method to consume a .wav file.
081         */
082        public void consume() throws IOException {
083            writer.consume();
084        }
085    
086        /**
087         * Method to return duration.
088         *
089         * @return duration in millis
090         */
091        @Override
092        public int getDuration() {
093            // http://social.msdn.microsoft.com/Forums/windows/en-US/5a92be69-3b4e-4d92-b1d2-141ef0a50c91/how-to-calculate-duration-of-wave-file-from-its-size?forum=winforms
094            double length = (double) getFile().length();
095            double sampleRate = (double) this.sampleRate;
096            double numChannels = (double) this.numChannels;
097            double bitsPerSample = (double) getBitsPerSample();
098    
099            double duration = (double) Math.round((length / (sampleRate * numChannels * bitsPerSample /8)) * 1000) / 1000 * 1000;
100            LoggerFactory.obtainLogger(TAG).
101                    d(String.format("getDuration# calculated duration is %s", duration));
102    
103            return (int) duration;
104        }

125            return 1;

129            return numChannels;
130        }
131    
132        int getSampleRate() {
133            return sampleRate;
134        }
135    
136        int getByteRate() {
137            if (sampleRate == 0 || numChannels == 0 || mBitsPerSample == 0) {
138                throw new IllegalStateException(
139                        String.format("ByteRate can't be set before either SampleRate(%d) or NumChannels(%d) or BitsPerSample(%d) are set!",
140                                sampleRate, numChannels, mBitsPerSample));
141            }
142    
143            return sampleRate * numChannels * mBitsPerSample / 8;
144        }
145    
146        short getBlockAlign() {
147            if (numChannels == 0 || mBitsPerSample == 0) {
148                throw new IllegalStateException(
149                        String.format("ByteRate can't be set before either NumChannels(%d) or BitsPerSample(%d) are set!",
150                                numChannels, mBitsPerSample));
151            }
152    
153            return (short) (numChannels * mBitsPerSample / 8);
154        }
155    
156        short getBitsPerSample() {
157            return mBitsPerSample;
158        }

175        /**
176         * Returns arrays of bytes of both splitted parts.
177         *
178         * @param wav .wav file which data to split
179         * @param durationPlayed duration of the first part in millis
180         * @return arrays of bytes as of both splitted parts
181         */
182        @Override
183        public List<byte[]> getDataParts(SoundFile wav, int durationPlayed) {
184            double durationTotal = wav.getDuration();
185    
186            double percent = durationTotal / 100;
187            double alreadyPlayed = (double) Math.round(durationPlayed / percent * 100) / 100;
188    
189            wav.read();
190    
191            LoggerFactory.obtainLogger(TAG).
192                    d(String.format("split# data duration = %d", wav.getData().length));
193    
194            int d1 = (int) (wav.getData().length / 100 * alreadyPlayed);
195            LoggerFactory.obtainLogger(TAG).
196                    d(String.format("split# d1 = %d", d1));
197    
198            int d2 = wav.getData().length - d1;
199            LoggerFactory.obtainLogger(TAG).
200                    d(String.format("split# d2 = %d", d2));
201    
202            LoggerFactory.obtainLogger(TAG).
203                    d(String.format("split# (d1 + d2) = %d", (d1 + d2)));
204    
205            List<byte[]> parts = new ArrayList<byte[]>(2);
206    
207            parts.add(0, Arrays.copyOfRange(wav.getData(), 0, d1));
208            parts.add(1, Arrays.copyOfRange(wav.getData(), d1, wav.getData().length - 1));
209    
210            return parts;
211        }

218                }

221                }
222            };

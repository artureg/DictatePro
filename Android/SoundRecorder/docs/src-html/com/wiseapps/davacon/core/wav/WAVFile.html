067        private WAVFile(File file, int audioFormat, int channelConfig, int sampleRateInHz) throws IOException {
068            if (audioFormat != AudioFormat.ENCODING_PCM_16BIT) {
069                throw new IllegalArgumentException("Incorrect audio format, working with PCM 16 bit only!");
070            }
071    
072            setChunkID(DEFAULT_CHUNK_ID);
073            setChunkSize(DEFAULT_CHUNK_SIZE);
074            setFormat(DEFAULT_FORMAT);
075    
076            setSubchunk1ID(DEFAULT_SUBCHUNK_1_ID);
077            setSubchunk1Size(DEFAULT_SUBCHUNK_1_SIZE);
078            setAudioFormat(DEFAULT_AUDIO_FORMAT);
079            setNumChannels((short) (channelConfig == AudioFormat.CHANNEL_IN_STEREO ? 2 : 1));
080            setSampleRate(sampleRateInHz);
081            setByteRate();
082            setBlockAlign();
083            setBitsPerSample(DEFAULT_BITS_PER_SAMPLE);
084    
085            setSubchunk2ID(DEFAULT_SUBCHUNK_2_ID);
086            setSubchunk2Size(DEFAULT_SUBCHUNK_2_SIZE);
087    
088            setFile(file);
089    
090            writer = new WAVFileWriter(this);
091            reader = new WAVFileReader(this);
092        }

118        /**
119         * Method to return duration.
120         *
121         * @return duration in millis
122         */
123        @Override
124        public int getDuration() {
125            // http://social.msdn.microsoft.com/Forums/windows/en-US/5a92be69-3b4e-4d92-b1d2-141ef0a50c91/how-to-calculate-duration-of-wave-file-from-its-size?forum=winforms
126            double length = (double) getFile().length();
127            double sampleRate = (double) this.sampleRate;
128            double numChannels = (double) this.numChannels;
129            double bitsPerSample = (double) getBitsPerSample();
130    
131            double duration = (double) Math.round((length / (sampleRate * numChannels * bitsPerSample /8)) * 1000) / 1000 * 1000;
132            LoggerFactory.obtainLogger(TAG).
133                    d(String.format("getDuration# calculated duration is %s", duration));
134    
135            return (int) duration;
136        }

200        private void setByteRate() {
201            if (sampleRate == 0 || numChannels == 0 || bitsPerSample == 0) {
202                LoggerFactory.obtainLogger(TAG).
203                        d(String.format("Actual byte rate can't be set before either " +
204                                "SampleRate(%d) or NumChannels(%d) or BitsPerSample(%d) are set!",
205                                sampleRate, numChannels, bitsPerSample));
206                LoggerFactory.obtainLogger(TAG).
207                        d(String.format("Setting defaults (%s).", DEFAULT_BYTE_RATE));
208                this.byteRate = DEFAULT_BYTE_RATE;
209    
210                return;
211            }
212    
213            this.byteRate = sampleRate * numChannels * bitsPerSample / 8;
214        }

222        private void setBlockAlign() {
223            if (numChannels == 0 || bitsPerSample == 0) {
224                LoggerFactory.obtainLogger(TAG).
225                        d(String.format("Block align can't be set before either NumChannels(%d) or BitsPerSample(%d) are set!",
226                                numChannels, bitsPerSample));
227                this.blockAlign = DEFAULT_BLOCK_ALIGN;
228    
229                return;
230            }
231    
232            this.blockAlign = (short) (numChannels * bitsPerSample / 8);
233        }

273        /**
274         * Returns arrays of bytes of both splitted parts.
275         *
276         * @param wav .wav file which binary content to split
277         * @param durationPlayed duration of the first part in millis
278         * @return arrays of bytes as of both splitted parts
279         */
280        @Override
281        public List<byte[]> getDataParts(SoundFile wav, int durationPlayed) {
282            double durationTotal = wav.getDuration();
283    
284            double percent = durationTotal / 100;
285            double alreadyPlayed = (double) Math.round(durationPlayed / percent * 100) / 100;
286    
287            wav.read();
288    
289            LoggerFactory.obtainLogger(TAG).
290                    d(String.format("split# data duration = %d", wav.getData().length));
291    
292            int d1 = (int) (wav.getData().length / 100 * alreadyPlayed);
293            LoggerFactory.obtainLogger(TAG).
294                    d(String.format("split# d1 = %d", d1));
295    
296            int d2 = wav.getData().length - d1;
297            LoggerFactory.obtainLogger(TAG).
298                    d(String.format("split# d2 = %d", d2));
299    
300            LoggerFactory.obtainLogger(TAG).
301                    d(String.format("split# (d1 + d2) = %d", (d1 + d2)));
302    
303            List<byte[]> parts = new ArrayList<byte[]>(2);
304    
305            parts.add(0, Arrays.copyOfRange(wav.getData(), 0, d1));
306            parts.add(1, Arrays.copyOfRange(wav.getData(), d1, wav.getData().length - 1));
307    
308            return parts;
309        }

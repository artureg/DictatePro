020         * Concatenates the tracks.
021         *
022         * @param context Context
023         * @param wavs objects of {@link com.wiseapps.davacon.core.SoundFile} type to concatenate
024         * @param filename name of the resultant file
025         * @return the object of {@link com.wiseapps.davacon.core.SoundFile} type with the concatenated contents
026         * @throws Exception
027         */
028        public static SoundFile concat(Context context, List<SoundFile> wavs, String filename) throws Exception {
029            SoundFile wav = SoundFile.create(new File(filename));
030    
031            for (SoundFile w : wavs) {
032                w.read();
033                wav.write(w.getData());
034            }
035    
036            wav.consume();

052        /**
053         * Splits the track.
054         *
055         * @param context Context
056         * @param wav object of {@link com.wiseapps.davacon.core.SoundFile} to split the contents of
057         * @param durationPlayed duration of the first part in millis
058         * @throws Exception
059         */
060        public static void split(Context context, SoundFile wav, int durationPlayed) throws Exception {
061            List<byte[]> parts = wav.getDataParts(wav, durationPlayed);
062    
063            SoundFile wav1 = SoundFile.create(new File(FileUtils.getFilename(context)));
064            wav1.write(parts.get(0));
065            wav1.consume();
066    
067            SoundFile wav2 = SoundFile.create(new File(FileUtils.getFilename(context)));
068            wav2.write(parts.get(1));
069            wav2.consume();
070        }

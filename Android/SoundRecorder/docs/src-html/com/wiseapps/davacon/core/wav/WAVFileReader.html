001    package com.wiseapps.davacon.core.wav;
002    
003    import com.wiseapps.davacon.logging.LoggerFactory;
004    
005    import java.io.FileInputStream;
006    import java.io.FileNotFoundException;
007    import java.io.IOException;
008    
009    /**
010     * Class to read .wav file.
011     *
012     * @author varya.bzhezinskaya@gmail.com
013     *         Date: 3/30/14
014     *         Time: 11:35 AM
015     */
016    public class WAVFileReader {
017        private static final String TAG = WAVFileReader.class.getSimpleName();
018    
019        private FileInputStream stream;
020    
021        private final WAVFile wav;
022    
023        public WAVFileReader(WAVFile wav) {
024            this.wav = wav;
025        }
026    
027        /**
028         * Reads .wav file.
029         */
030        public void read() {
031            try {
032                stream = new FileInputStream(wav.getFile());
033    
034                if (!prepare()) {
035                    LoggerFactory.obtainLogger(TAG).
036                            d("read# .wav file has incorrect format, returning...");
037                    return;
038                }
039    
040                int length = (int) wav.getFile().length() - 44;
041                LoggerFactory.obtainLogger(TAG).
042                        d(String.format("read# file data length is %d", length));
043    
044                byte[] data = new byte[length];
045                stream.read(data, 0, length);
046    
047                wav.setData(data);
048    
049                LoggerFactory.obtainLogger(TAG).
050                        d(String.format("read# data lenght is %d", data.length));
051            } catch (FileNotFoundException e) {
052                LoggerFactory.obtainLogger(TAG).e(
053                        String.format("File %s can not be found!", wav.getFile().getAbsolutePath()),
054                        e);
055            } catch (IOException e) {
056                LoggerFactory.obtainLogger(TAG).e(
057                        String.format("Error while reading %s", wav.getFile().getAbsolutePath()),
058                        e);
059            } finally {
060                if (stream != null) {
061                    try {
062                        stream.close();
063                    } catch (IOException e) {
064                        LoggerFactory.obtainLogger(TAG).e(e.getMessage(), e);
065                    }
066                }
067            }
068        }
069    
070        private boolean prepare()throws IOException {
071            long length = wav.getFile().length();
072            LoggerFactory.obtainLogger(TAG).
073                    d(String.format("prepare# .wav file length is %s", length));
074    
075            // ChunkID
076            byte[] data = new byte[4];
077            stream.read(data, 0, 4);
078            if (data[0] != 'R' ||
079                    data[1] != 'I' ||
080                    data[2] != 'F' ||
081                    data[3] != 'F') {
082                LoggerFactory.obtainLogger(TAG).d("prepare# incorrect ChunkID");
083                return false;
084            }
085            wav.setChunkID(new String(data));
086    
087            // ChunkSize
088            int chunkSize = readInt();
089            LoggerFactory.obtainLogger(TAG).
090                    d(String.format("prepare# ChunkSize = %s", chunkSize));
091            wav.setChunkSize(chunkSize);
092    
093            // Format
094            data = new byte[4];
095            stream.read(data, 0, 4);
096            if (data[0] != 'W' ||
097                    data[1] != 'A' ||
098                    data[2] != 'V' ||
099                    data[3] != 'E') {
100                LoggerFactory.obtainLogger(TAG).d("prepare# incorrect Format");
101                return false;
102            }
103            wav.setFormat(new String(data));
104    
105            // Subchunk1ID
106            data = new byte[4];
107            stream.read(data, 0, 4);
108            if (data[0] != 'f' ||
109                    data[1] != 'm' ||
110                    data[2] != 't' ||
111                    data[3] != ' ' ) {
112                LoggerFactory.obtainLogger(TAG).d("prepare# incorrect Subchunk1ID");
113                return false;
114            }
115            wav.setSubchunk1ID(new String(data));
116    
117            // Subchunk1Size
118            int subchunk1Size = readInt();
119            LoggerFactory.obtainLogger(TAG).
120                    d(String.format("prepare# Subchunk1Size = %s", subchunk1Size));
121            wav.setSubchunk1Size(subchunk1Size);
122    
123            // AudioFormat
124            short audioFormat = readShort();
125            LoggerFactory.obtainLogger(TAG).
126                    d(String.format("prepare# AudioFormat = %s", audioFormat));
127            wav.setAudioFormat(audioFormat);
128    
129            // NumChannels
130            short numChannels = readShort();
131            LoggerFactory.obtainLogger(TAG).
132                    d(String.format("prepare# NumChannels = %s", numChannels));
133            wav.setNumChannels(numChannels);
134    
135            // SampleRate
136            int sampleRate = readInt();
137            LoggerFactory.obtainLogger(TAG).
138                    d(String.format("prepare# SampleRate = %s", sampleRate));
139            wav.setSampleRate(sampleRate);
140    
141            // ByteRate
142            int byteRate = readInt();
143            LoggerFactory.obtainLogger(TAG).
144                    d(String.format("prepare# ByteRate = %s", byteRate));
145            wav.setByteRate(byteRate);
146    
147            // BlockAlign
148            short blockAlign = readShort();
149            LoggerFactory.obtainLogger(TAG).
150                    d(String.format("prepare# BlockAlign = %s", blockAlign));
151            wav.setBlockAlign(blockAlign);
152    
153            // BitsPerSample
154            short bitsPerSample = readShort();
155            LoggerFactory.obtainLogger(TAG).
156                    d(String.format("prepare# BitsPerSample = %s", bitsPerSample));
157            wav.setBitsPerSample(bitsPerSample);
158    
159            // Subchunk2ID
160            data = new byte[4];
161            stream.read(data, 0, 4);
162            if (data[0] != 'd' ||
163                    data[1] != 'a' ||
164                    data[2] != 't' ||
165                    data[3] != 'a' ) {
166                LoggerFactory.obtainLogger(TAG).d("prepare# incorrect Subchunk2ID");
167                return false;
168            }
169            wav.setSubchunk2ID(new String(data));
170    
171            // Subchunk2Size
172            int subchunk2Size = readInt();
173            LoggerFactory.obtainLogger(TAG).
174                    d(String.format("prepare# Subchunk2Size = %s", subchunk2Size));
175            wav.setSubchunk2Size(subchunk2Size);
176    
177            LoggerFactory.obtainLogger(TAG).
178                    d(String.format("prepare# chunkSize - subchunk2Size = %s", (chunkSize - subchunk2Size)));
179            LoggerFactory.obtainLogger(TAG).
180                    d(String.format("prepare# .wav file length - chunkSize = %s", (length - chunkSize)));
181    
182            return wav.isFormatCorrect();
183        }
184    
185        private short readShort() throws IOException {
186            byte[] data = new byte[2];
187            stream.read(data, 0, 2);
188            return (short)(((0xff & data[1]) << 8) |
189                    ((0xff & data[0])));
190        }
191    
192        private int readInt() throws IOException {
193            byte[] data = new byte[4];
194            stream.read(data, 0, 4);
195            return ((0xff & data[3]) << 24) |
196                    ((0xff & data[2]) << 16) |
197                    ((0xff & data[1]) << 8) |
198                    ((0xff & data[0]));
199        }
200    }

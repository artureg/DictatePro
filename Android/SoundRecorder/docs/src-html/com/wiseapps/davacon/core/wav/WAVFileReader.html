001    package com.wiseapps.davacon.core.wav;
002    
003    import com.wiseapps.davacon.logging.LoggerFactory;
004    
005    import java.io.FileInputStream;
006    import java.io.FileNotFoundException;
007    import java.io.IOException;
008    
009    /**
010     * Class to read .wav file.
011     *
012     * @author varya.bzhezinskaya@gmail.com
013     *         Date: 3/30/14
014     *         Time: 11:35 AM
015     */
016    public class WAVFileReader {
017        private static final String TAG = WAVFileReader.class.getSimpleName();
018    
019        private final WAVFile wav;
020    
021        public WAVFileReader(WAVFile wav) {
022            this.wav = wav;
023        }
024    
025        /**
026         * Reads .wav file.
027         */
028        public void read() {
029            FileInputStream stream = null;
030    
031            try {
032                stream = new FileInputStream(wav.getFile());
033    
034                // omit the header and both subchunks
035                stream.read(new byte[44], 0, 44);
036    
037                int length = (int) wav.getFile().length() - 44;
038                LoggerFactory.obtainLogger(TAG).
039                        d(String.format("read# file data lenght is %d", length));
040    
041                byte[] data = new byte[length];
042                stream.read(data, 0, length);
043    
044                wav.setData(data);
045    
046                LoggerFactory.obtainLogger(TAG).
047                        d(String.format("read# data lenght is %d", data.length));
048            } catch (FileNotFoundException e) {
049                LoggerFactory.obtainLogger(TAG).e(
050                        String.format("File %s can not be found!", wav.getFile().getAbsolutePath()),
051                        e);
052            } catch (IOException e) {
053                LoggerFactory.obtainLogger(TAG).e(
054                        String.format("Error while reading %s", wav.getFile().getAbsolutePath()),
055                        e);
056            } finally {
057                if (stream != null) {
058                    try {
059                        stream.close();
060                    } catch (IOException e) {
061                        LoggerFactory.obtainLogger(TAG).e(e.getMessage(), e);
062                    }
063                }
064            }
065        }
066    }

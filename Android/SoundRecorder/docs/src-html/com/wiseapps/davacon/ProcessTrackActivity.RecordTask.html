117        /**
118         * Helper method to initialize track.
119         */
120        private void initData() {
121            Bundle bundle = getIntent().getBundleExtra(BUNDLE);
122    
123            if (bundle != null) {
124                File track = (File) bundle.getSerializable(EXTRA_TRACK);
125                if (track == null) {
126                    return;
127                }
128    
129                try {
130                    sf = SoundFile.create(track);
131                } catch (IOException e) {
132                    LoggerFactory.obtainLogger(TAG).
133                            e(String.format("initData# Couldn't read %s", track.getAbsolutePath()), e);
134                }
135            }
136        }

165        /**
166         * Records the track.
167         *
168         * <p>In case the track already exists, reresords it.</p>
169         *
170         * @param view The clicked view.
171         */
172        public void record(View view) {
173            if (mRecorder == null) {
174    
175                if (startRecording()) {
176                    isRecording = true;
177    
178                    buttonRecord.setImageDrawable(getResources().
179                            getDrawable(R.drawable.ic_action_stop));
180                } else {
181                    Toast.makeText(this,
182                            getResources().getString(R.string.prompt_recording_failed), Toast.LENGTH_SHORT).show();
183                }
184            } else {
185                isRecording = false;
186    
187                buttonRecord.setImageDrawable(getResources().
188                        getDrawable(R.drawable.ic_action_mic));
189    
190                stopRecording();
191            }
192        }

194        /**
195         * Helper method to start recording.
196         *
197         * @return boolean Returns true if track recording has successfully started, false otherwise.
198         */
199        private boolean startRecording() {
200            mRecorder = new AudioRecord(RECORDER_AUDIO_SOURCE,
201                    RECORDER_SAMPLE_RATE_IN_HZ, RECORDER_CHANNEL_CONFIG, RECORDER_AUDIO_FORMAT, RECORDER_BUFFER_SIZE_IN_BYTES);
202            LoggerFactory.obtainLogger(TAG).d("startRecording# audioSource = " + mRecorder.getAudioSource());
203            LoggerFactory.obtainLogger(TAG).d("startRecording# sampleRateInHz = " + mRecorder.getSampleRate());
204            LoggerFactory.obtainLogger(TAG).d("startRecording# channelConfig = " + mRecorder.getChannelConfiguration());
205            LoggerFactory.obtainLogger(TAG).d("startRecording# audioFormat = " + mRecorder.getAudioFormat());
206            LoggerFactory.obtainLogger(TAG).d("startRecording# bufferSizeInBytes = " + RECORDER_BUFFER_SIZE_IN_BYTES);
207    
208            LoggerFactory.obtainLogger(TAG).d("startRecording# mRecorder.getState() = " + mRecorder.getState());
209    
210            if (mRecorder.getState() == STATE_INITIALIZED) {
211                mRecorder.startRecording();
212    
213                mTask = new RecordTask();
214                mTask.execute();
215    
216                return true;
217            }
218    
219            return false;
220        }
221    
222        /**
223         * Helper method to stop recording.
224         */
225        private void stopRecording() {
226            if (mRecorder.getState() == STATE_INITIALIZED) {
227                mRecorder.stop();
228            }
229    
230            mRecorder.release();
231            mRecorder = null;
232    
233            if (mTask != null) {
234                mTask.cancel(true);
235                mTask = null;
236            }
237        }

345        /**
346         * Async task to record a track.
347         */
348        private class RecordTask extends AsyncTask<Void, Integer, Void> {
349            private SoundFile sf;
350    
351            private double startSeconds = 0,
352                    endSeconds = 180;
353            private long currentTimeMillis;
354    
355            @Override
356            protected void onPreExecute() {
357                super.onPreExecute();
358    
359                currentTimeMillis = System.currentTimeMillis();
360    
361                textDuration.setText(
362                        String.format(getResources().getString(R.string.process_track_duration),
363                                DurationUtils.format(startSeconds * 1000), DurationUtils.format(endSeconds * 1000)));
364                textDuration.setVisibility(View.VISIBLE);
365    
366                progressBar.setMax((int) endSeconds);
367                progressBar.setVisibility(View.VISIBLE);
368            }
369    
370            @Override
371            protected Void doInBackground(Void... voids) {
372                try {
373                    // rewrite if exists
374                    sf = SoundFile.create(new File(ProcessTrackActivity.this.sf != null ?
375                            ProcessTrackActivity.this.sf.getFile().getAbsolutePath() : FileUtils.getFilename(getApplicationContext())));
376    
377                    byte data[] = new byte[RECORDER_BUFFER_SIZE_IN_BYTES];
378    
379                    int read;
380                    while(isRecording) {
381                        read = mRecorder.read(data, 0, RECORDER_BUFFER_SIZE_IN_BYTES);
382                        LoggerFactory.obtainLogger(TAG).d("RecordTask.doInBackground# data = " + data.length);
383                        LoggerFactory.obtainLogger(TAG).d("RecordTask.doInBackground# read result = " + read);
384    
385                        if (read != AudioRecord.ERROR_INVALID_OPERATION) {
386                            sf.write(data);
387                        }
388    
389                        long currentTimeMillis = System.currentTimeMillis();
390                        if (currentTimeMillis - this.currentTimeMillis > 100) {
391                            startSeconds += 0.1;
392                            this.currentTimeMillis = currentTimeMillis;
393    
394                            publishProgress((int) startSeconds);
395                        }
396                    }
397                } catch (Exception e) {
398                    LoggerFactory.obtainLogger(TAG).e(e.getMessage(), e);
399                }
400    
401                return null;
402            }
403    
404            @Override
405            protected void onProgressUpdate(Integer... values) {
406                super.onProgressUpdate(values);
407    
408                textDuration.setText(
409                        String.format(getResources().getString(R.string.process_track_duration),
410                                DurationUtils.format(startSeconds * 1000), DurationUtils.format(endSeconds * 1000)));
411                progressBar.setProgress((int)startSeconds);
412            }
413    
414            @Override
415            protected void onCancelled() {
416                super.onCancelled();
417    
418                if (sf != null) {
419                    try {
420                        sf.consume();
421                    } catch (IOException e) {
422                        LoggerFactory.obtainLogger(TAG).e(e.getMessage(), e);
423    
424                        textDuration.setText("");
425                        textDuration.setVisibility(View.GONE);
426                    }
427    
428                    ProcessTrackActivity.this.sf = sf;
429    
430                    buttonPlay.setImageDrawable(getResources().
431                            getDrawable(R.drawable.ic_action_play));
432                    buttonPlay.setVisibility(View.VISIBLE);
433    
434                    textDuration.setText(
435                            String.format(getResources().getString(R.string.process_track_duration),
436                                    DurationUtils.format(0), DurationUtils.format(sf.getDuration())));
437    
438                    progressBar.setVisibility(View.GONE);
439                }
440            }
441        }

443        /**
444         * Async task to split a track.
445         */
446        private class SplitTask extends AsyncTask<Void, Void, Boolean> {
447            @Override
448            protected Boolean doInBackground(Void... voids) {
449                try {
450                    SoundFileHandler.split(
451                            ProcessTrackActivity.this, sf, getCurrentPosition());
452                    return true;
453                } catch (Exception e) {
454                    LoggerFactory.obtainLogger(TAG).e(e.getMessage(), e);
455                    return false;
456                }
457            }
458    
459            @Override
460            protected void onPostExecute(Boolean result) {
461                super.onPostExecute(result);
462    
463                if (!result) {
464                    Toast.makeText(ProcessTrackActivity.this,
465                            getResources().getString(R.string.prompt_split_failed), Toast.LENGTH_SHORT).show();
466                    return;
467                }
468    
469                if (sf.getFile().delete()) {
470                    LoggerFactory.obtainLogger(TAG).
471                            d(String.format("onPostExecute# File %s deleted successully",
472                                    sf.getFile().getAbsolutePath()));
473                } else {
474                    LoggerFactory.obtainLogger(TAG).
475                            d(String.format("onPostExecute# File %s deletion failed",
476                                    sf.getFile().getAbsolutePath()));
477                }
478    
479                setResult(Activity.RESULT_OK);
480                finish();
481            }
482        }

082        /**
083         * Performs final cleanup before an activity is destroyed.
084         * <p></>Deletes any temporary files if there are any.<p/>
085         */
086        @Override
087        protected void onDestroy() {
088            File[] tracks = FileUtils.getRoot(this).listFiles();
089    
090            if (tracks != null) {
091                for (File track : tracks) {
092                    if (track.getName().contains(TMP_SUFFIX)) {
093                        boolean result = track.delete();
094                        LoggerFactory.obtainLogger(TAG).
095                                d(String.format("onDestroy# File %s %s",
096                                        track.getAbsolutePath(), result ? "deleted successully" : "deletion failed"));
097                    }
098                }
099            }
100    
101            super.onDestroy();
102        }

104        /**
105         * Called when an activity you launched exits, giving you the requestCode
106         * you started it with, the resultCode it returned, and any additional
107         * data from it.  The <var>resultCode</var> will be
108         * {@link #RESULT_CANCELED} if the activity explicitly returned that,
109         * didn't return any result, or crashed during its operation.
110         *
111         * In case of <var>requestCode</var> equal to REQUEST_CODE_PROCESS_TRACK (one is returned from the
112         * {@link com.wiseapps.davacon.ProcessTrackActivity ProcessTrackActivity}) and
113         * <var>resultCode</var> equal to Activity.RESULT_OK, class fields are reinitializaed, UI is updated.
114         *
115         * @param requestCode The integer request code originally supplied to
116         *                    startActivityForResult(), allowing you to identify who this
117         *                    result came from.
118         * @param resultCode The integer result code returned by the child activity
119         *                   through its setResult().
120         * @param data An Intent, which can return result data to the caller
121         *               (various data can be attached to Intent "extras").
122         */
123        @Override
124        protected void onActivityResult(int requestCode, int resultCode, Intent data) {
125            super.onActivityResult(requestCode, resultCode, data);
126    
127            switch(requestCode) {
128                case REQUEST_CODE_PROCESS_TRACK: {
129                    if (resultCode == Activity.RESULT_OK) {
130                        initData();
131                        updateWidgets();
132                    }
133    
134                    break;
135                }
136            }
137        }

160        /**

162         *

164         * to provide track deletion functionality.<p/>
165         *
166         * <p>In case Add menu is called the user id forwarded to the
167         * {@link com.wiseapps.davacon.ProcessTrackActivity ProcessTrackActivity}
168         * where a track can be recorded.<p/>
169         *
170         * @param item The menu item that was selected.
171         * @return boolean Return false to allow normal menu processing to
172         *         proceed, true to consume it here.
173         */
174        @Override
175        public boolean onOptionsItemSelected(MenuItem item) {
176            switch (item.getItemId()) {
177                case R.id.edit: {
178                    mode = Mode.EDIT;
179    
180                    updateWidgets();
181    
182                    return true;
183                }
184                case R.id.add: {
185    //                LoggerFactory.obtainLogger(TAG).
186    //                        d("onOptionsItemSelected# test jni " + SpeexWrapper.test(2, 3));
187                    ActivityNavigator.startProcessTrackActivityForResult(this, REQUEST_CODE_PROCESS_TRACK);
188                    return true;
189                }
190            }
191    
192            return false;
193        }

195        /**
196         * Helper method to initialize tracks list.
197         */
198        private void initData() {
199            this.sfs = null;
200    
201            File root = FileUtils.getRoot(this);
202    
203            File[] tracks = root.listFiles();
204            if (tracks == null || tracks.length == 0) {
205                return;
206            }
207    
208            this.sfs = new ArrayList<SoundFile>();
209            for (File track : tracks) {
210                if (track.getName().contains(TMP_SUFFIX)) {
211                    continue;
212                }
213                try {
214                    this.sfs.add(SoundFile.create(track));
215                } catch (IOException e) {
216                    LoggerFactory.obtainLogger(TAG).
217                            e(String.format("initData# Couldn't read %s", track.getAbsolutePath()), e);
218                }
219            }
220        }

249        /**
250         * Helper method to update menu in case screen's UI mode has been changed.
251         */
252        private void updateMenu() {
253            if (menuEdit == null) {
254                return;
255            }
256    
257            // set icon
258            menuEdit.setIcon(mode == Mode.VIEW ? getResources().getDrawable(R.drawable.ic_action_edit) :
259                    getResources().getDrawable(R.drawable.ic_action_accept));
260    
261            // set visibility
262            menuEdit.setVisible(sfs != null && !sfs.isEmpty());
263        }

265        /**
266         * Helper method to update tracks list in case screen's UI mode has been changed.
267         */
268        private void updateTrackList() {
269            if (sfs == null || sfs.isEmpty()) {
270                textDuration.setVisibility(View.GONE);
271                trackList.setVisibility(View.GONE);
272                return;
273            }
274    
275            double overAllDuration = 0;
276    
277            LayoutInflater inflater = getLayoutInflater();
278            trackList.removeAllViews();
279    
280            int count = 0;
281            double duration;
282    
283            View convertView;
284            for (final SoundFile sf : sfs) {
285                convertView = inflater.inflate(R.layout.track, null);
286    
287                duration = DurationUtils.format(sf.getDuration());
288                overAllDuration += duration;
289    
290                ((TextView) convertView.findViewById(R.id.track)).
291                        setText(String.format(getResources().getString(R.string.track_duration),
292                                duration));
293    
294                switch(mode) {
295                    case VIEW: {
296                        convertView.findViewById(R.id.action_forward).setVisibility(View.VISIBLE);
297                        convertView.setOnClickListener(new View.OnClickListener() {
298                            @Override
299                            public void onClick(View view) {
300                                onDetails(sf);
301                            }
302                        });
303    
304                        break;
305                    }
306                    case EDIT: {
307                        convertView.findViewById(R.id.action_remove).setVisibility(View.VISIBLE);
308                        convertView.findViewById(R.id.action_remove).setOnClickListener(new View.OnClickListener() {
309                            @Override
310                            public void onClick(View view) {
311                                onDelete(sf);
312                            }
313                        });
314    
315                        break;
316                    }
317                }
318    
319                ((LinearLayout) findViewById(R.id.tracks)).addView(convertView);
320    
321                if (count < sfs.size()) {
322                    trackList.addView(inflater.inflate(R.layout.separator, null));
323                }
324    
325                ++count;
326            }
327    
328            trackList.setVisibility(View.VISIBLE);
329    
330            textDuration.setText(String.format(getResources().getString(R.string.main_duration),
331                    DurationUtils.format(0), overAllDuration));
332            textDuration.setVisibility(View.VISIBLE);
333        }

348        /**
349         * Clears the tracks list.
350         *
351         * @param view The clicked view.
352         */
353        public void clearAll(View view) {
354            File root = FileUtils.getRoot(this);
355            File[] tracks = root.listFiles();
356    
357            if (tracks != null && tracks.length != 0) {
358                for (File track : tracks) {
359                    boolean result = track.delete();
360                    LoggerFactory.obtainLogger(TAG).
361                            d(String.format("clearAll# File %s %s",
362                                    track.getAbsolutePath(), result ? "deleted successully" : "deletion failed"));
363                }
364            }
365    
366            initData();
367            updateWidgets();
368        }

470        public void onDelete(SoundFile wav) {
471            if (wav.getFile().delete()) {
472                LoggerFactory.obtainLogger(TAG).
473                        d(String.format("onPostExecute# File %s deleted successully",
474                                wav.getFile().getAbsolutePath()));
475    
476                Toast.makeText(MainActivity.this,
477                        getResources().getString(R.string.prompt_file_deleted), Toast.LENGTH_SHORT).show();
478            } else {
479                LoggerFactory.obtainLogger(TAG).
480                        d(String.format("onPostExecute# File %s deletion failed",
481                                wav.getFile().getAbsolutePath()));
482            }
483    
484            mode = Mode.VIEW;
485    
486            initData();
487            updateWidgets();
488        }

490        /**
491         * Async task to play the tracks in succession.
492         */
493        private class PlayAllTask extends AsyncTask<Void, Void, Boolean> {
494            @Override
495            protected Boolean doInBackground(Void... voids) {
496                try {
497                    tmp = SoundFileHandler.concat(MainActivity.this, sfs);
498                    LoggerFactory.obtainLogger(TAG).
499                            d(String.format("doInBackground# Tmp file %s created successfully", tmp.getFile().getAbsolutePath()));
500    
501                    return true;
502                } catch (Exception e) {
503                    LoggerFactory.obtainLogger(TAG).
504                            e("Tmp file creation failed", e);
505                    return false;
506                }
507            }
508    
509            @Override
510            protected void onPostExecute(Boolean result) {
511                super.onPostExecute(result);
512    
513                if (!result) {
514                    Toast.makeText(MainActivity.this,
515                            getResources().getString(R.string.prompt_concat_failed), Toast.LENGTH_SHORT).show();
516                    return;
517                }
518    
519                doPlay();
520            }
521        }

001    package com.wiseapps.davacon;
002    
003    import android.app.Activity;
004    import android.content.Intent;
005    import android.os.Bundle;
006    
007    /**
008     * Helper class to provide navigation between the application screens.
009     *
010     * @author varya.bzhezinskaya@gmail.com
011     *         Date: 3/20/14
012     *         Time: 12:13 PM
013     */
014    public class ActivityNavigator {
015    
016        public static final String BUNDLE = "bundle";
017    
018        public static final String EXTRA_TRACK = "track";
019    
020        public static void startProcessTrackActivityForResult(Activity activity, int requestCode) {
021            startProcessTrackActivityForResult(activity, requestCode, null);
022        }
023    
024        public static void startProcessTrackActivityForResult(Activity activity, int requestCode, Bundle bundle) {
025            Intent intent = new Intent(activity, ProcessTrackActivity.class);
026    
027            if (bundle != null) {
028                intent.putExtra(BUNDLE, bundle);
029            }
030    
031            activity.startActivityForResult(intent, requestCode);
032        }
033    }

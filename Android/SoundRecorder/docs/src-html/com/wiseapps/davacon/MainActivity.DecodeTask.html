084        /**
085         * Performs final cleanup before an activity is destroyed.
086         * <p></>Deletes any temporary files if there are any.<p/>
087         */
088        @Override
089        protected void onDestroy() {
090            File[] tracks = FileUtils.getRoot(this).listFiles();
091    
092            if (tracks != null) {
093                for (File track : tracks) {
094                    if (track.getName().contains(TMP_SUFFIX)) {
095                        boolean result = track.delete();
096                        LoggerFactory.obtainLogger(TAG).
097                                d(String.format("onDestroy# File %s %s",
098                                        track.getAbsolutePath(), result ? "deleted successully" : "deletion failed"));
099                    }
100                }
101            }
102    
103            super.onDestroy();
104        }

106        /**
107         * Called when an activity you launched exits, giving you the requestCode
108         * you started it with, the resultCode it returned, and any additional
109         * data from it.  The <var>resultCode</var> will be
110         * {@link #RESULT_CANCELED} if the activity explicitly returned that,
111         * didn't return any result, or crashed during its operation.
112         *
113         * In case of <var>requestCode</var> equal to REQUEST_CODE_PROCESS_TRACK (one is returned from the
114         * {@link com.wiseapps.davacon.ProcessTrackActivity ProcessTrackActivity}) and
115         * <var>resultCode</var> equal to Activity.RESULT_OK, class fields are reinitializaed, UI is updated.
116         *
117         * @param requestCode The integer request code originally supplied to
118         *                    startActivityForResult(), allowing you to identify who this
119         *                    result came from.
120         * @param resultCode The integer result code returned by the child activity
121         *                   through its setResult().
122         * @param data An Intent, which can return result data to the caller
123         *               (various data can be attached to Intent "extras").
124         */
125        @Override
126        protected void onActivityResult(int requestCode, int resultCode, Intent data) {
127            super.onActivityResult(requestCode, resultCode, data);
128    
129            switch(requestCode) {
130                case REQUEST_CODE_PROCESS_TRACK: {
131                    if (resultCode == Activity.RESULT_OK) {
132                        initData();
133                        updateWidgets();
134                    }
135    
136                    break;
137                }
138            }
139        }

167        /**
168         * Called whenever an item in the options menu is selected.
169         *
170         * <p>In case Edit menu is called the tracks list is updated
171         * to provide track deletion functionality.<p/>
172         *
173         * <p>In case Add menu is called the user id forwarded to the
174         * {@link com.wiseapps.davacon.ProcessTrackActivity ProcessTrackActivity}
175         * where a track can be recorded.<p/>
176         *
177         * @param item The menu item that was selected.
178         * @return boolean Return false to allow normal menu processing to
179         *         proceed, true to consume it here.
180         */
181        @Override
182        public boolean onOptionsItemSelected(MenuItem item) {
183            switch (item.getItemId()) {
184                case R.id.edit: {
185                    mode = Mode.EDIT;
186    
187                    updateWidgets();
188    
189                    return true;
190                }
191                case R.id.add: {
192    //                LoggerFactory.obtainLogger(TAG).
193    //                        d("onOptionsItemSelected# test jni " + SpeexWrapper.test(2, 3));
194                    ActivityNavigator.startProcessTrackActivityForResult(this, REQUEST_CODE_PROCESS_TRACK);
195                    return true;
196                }
197                case R.id.encode: {
198                    new EncodeTask().execute();
199                    return true;
200                }
201                case R.id.decode: {
202                    new DecodeTask().execute();
203                    return true;
204                }
205            }
206    
207            return false;
208        }

210        /**
211         * Helper method to initialize tracks list.
212         */
213        private void initData() {
214            this.sfs = null;
215    
216            File root = FileUtils.getRoot(this);
217    
218            File[] tracks = root.listFiles();
219            if (tracks == null || tracks.length == 0) {
220                return;
221            }
222    
223            this.sfs = new ArrayList<SoundFile>();
224            for (File track : tracks) {
225                if (track.getName().contains(TMP_SUFFIX) || track.getName().contains(SPEEX_SUFFIX)) {
226                    continue;
227                }
228                try {
229                    this.sfs.add(SoundFile.create(track));
230                } catch (IOException e) {
231                    LoggerFactory.obtainLogger(TAG).
232                            e(String.format("initData# Couldn't read %s", track.getAbsolutePath()), e);
233                }
234            }
235        }

282        /**
283         * Helper method to update tracks list in case screen's UI mode has been changed.
284         */
285        private void updateTrackList() {
286            if (sfs == null || sfs.isEmpty()) {
287                textDuration.setVisibility(View.GONE);
288                trackList.setVisibility(View.GONE);
289                return;
290            }
291    
292            double overAllDuration = 0;
293    
294            LayoutInflater inflater = getLayoutInflater();
295            trackList.removeAllViews();
296    
297            int count = 0;
298            double duration;
299    
300            View convertView;
301            for (final SoundFile sf : sfs) {
302                convertView = inflater.inflate(R.layout.track, null);
303    
304                duration = DurationUtils.format(sf.getDuration());
305                overAllDuration += duration;
306    
307                ((TextView) convertView.findViewById(R.id.track)).
308                        setText(String.format(getResources().getString(R.string.track_duration),
309                                duration));
310    
311                switch(mode) {
312                    case VIEW: {
313                        convertView.findViewById(R.id.action_forward).setVisibility(View.VISIBLE);
314                        convertView.setOnClickListener(new View.OnClickListener() {
315                            @Override
316                            public void onClick(View view) {
317                                onDetails(sf);
318                            }
319                        });
320    
321                        break;
322                    }
323                    case EDIT: {
324                        convertView.findViewById(R.id.action_remove).setVisibility(View.VISIBLE);
325                        convertView.findViewById(R.id.action_remove).setOnClickListener(new View.OnClickListener() {
326                            @Override
327                            public void onClick(View view) {
328                                onDelete(sf);
329                            }
330                        });
331    
332                        break;
333                    }
334                }
335    
336                ((LinearLayout) findViewById(R.id.tracks)).addView(convertView);
337    
338                if (count < sfs.size()) {
339                    trackList.addView(inflater.inflate(R.layout.separator, null));
340                }
341    
342                ++count;
343            }
344    
345            trackList.setVisibility(View.VISIBLE);
346    
347            textDuration.setText(String.format(getResources().getString(R.string.main_duration),
348                    DurationUtils.format(0), overAllDuration));
349            textDuration.setVisibility(View.VISIBLE);
350        }

487        public void onDelete(SoundFile wav) {
488            if (wav.getFile().delete()) {
489                LoggerFactory.obtainLogger(TAG).
490                        d(String.format("onPostExecute# File %s deleted successully",
491                                wav.getFile().getAbsolutePath()));
492    
493                Toast.makeText(MainActivity.this,
494                        getResources().getString(R.string.prompt_file_deleted), Toast.LENGTH_SHORT).show();
495            } else {
496                LoggerFactory.obtainLogger(TAG).
497                        d(String.format("onPostExecute# File %s deletion failed",
498                                wav.getFile().getAbsolutePath()));
499            }
500    
501            mode = Mode.VIEW;
502    
503            initData();
504            updateWidgets();
505        }

507        /**
508         * Async task to play the tracks in succession.
509         */
510        private class PlayAllTask extends AsyncTask<Void, Void, Boolean> {
511            @Override
512            protected Boolean doInBackground(Void... voids) {
513                try {
514                    tmp = SoundFileHandler.concat(MainActivity.this, sfs,
515                            FileUtils.getTempFilename(MainActivity.this, String.valueOf(System.currentTimeMillis()) + ".wav"));
516                    LoggerFactory.obtainLogger(TAG).
517                            d(String.format("doInBackground# Tmp file %s created successfully", tmp.getFile().getAbsolutePath()));
518    
519                    return true;
520                } catch (Exception e) {
521                    LoggerFactory.obtainLogger(TAG).
522                            e("Tmp file creation failed", e);
523                    return false;
524                }
525            }
526    
527            @Override
528            protected void onPostExecute(Boolean result) {
529                super.onPostExecute(result);
530    
531                if (!result) {
532                    Toast.makeText(MainActivity.this,
533                            getResources().getString(R.string.prompt_concat_failed), Toast.LENGTH_SHORT).show();
534                    return;
535                }
536    
537                doPlay();
538            }
539        }

541        private class EncodeTask extends AsyncTask<Void, Void, Boolean> {
542            @Override
543            protected Boolean doInBackground(Void... params) {
544                try {
545                    long currentTimeMillis = System.currentTimeMillis();
546                    speexTmp1 = SoundFileHandler.concat(MainActivity.this, sfs,
547                            FileUtils.getTempFilename(MainActivity.this, String.valueOf(currentTimeMillis) + ".wav"));
548                    LoggerFactory.obtainLogger(TAG).
549                            d(String.format("doInBackground# Speex file %s created successfully", speexTmp1.getFile().getAbsolutePath()));
550    
551                    speex = SoundFile.create(new File(FileUtils.getSpeexFilename(MainActivity.this,
552                            String.valueOf(currentTimeMillis) + ".wav")));
553                    int result = SpeexWrapper.encode(speexTmp1.getFile().getAbsolutePath(),
554                            speex.getFile().getAbsolutePath());
555    
556                    return result == 0;
557                } catch (Exception e) {
558                    LoggerFactory.obtainLogger(TAG).
559                            e("Speex file creation failed", e);
560                    return false;
561                }
562            }
563    
564            @Override
565            protected void onPostExecute(Boolean result) {
566                super.onPostExecute(result);
567    
568                if (!result) {
569                    Toast.makeText(MainActivity.this,
570                            getResources().getString(R.string.prompt_file_encoding_failed), Toast.LENGTH_SHORT).show();
571                    return;
572                }
573    
574                Toast.makeText(MainActivity.this,
575                        String.format(getResources().getString(R.string.prompt_file_encoded_successfully),
576                                speex.getFile().getAbsolutePath()),
577                        Toast.LENGTH_SHORT).show();
578            }
579        }

581        private class DecodeTask extends AsyncTask<Void, Void, Boolean> {
582            @Override
583            protected Boolean doInBackground(Void... params) {
584                try {
585                    if (speex == null) {
586                        LoggerFactory.obtainLogger(TAG).
587                                d("Initial speex file to decode not found");
588                        return false;
589                    }
590    
591                    String filename = FileUtils.getFilename(MainActivity.this,
592                            FileUtils.getFilenameFromSpeex(speex.getFile().getName()) + "-decoded.wav");
593    
594                    int result = SpeexWrapper.decode(speex.getFile().getAbsolutePath(), filename);
595                    if (result == 0) {
596                        speexTmp2 = SoundFile.create(new File(filename));
597                    }
598    
599                    return result == 0;
600                } catch (IOException e) {
601                    e.printStackTrace();
602    
603                    return false;
604                }
605            }
606    
607            @Override
608            protected void onPostExecute(Boolean result) {
609                super.onPostExecute(result);
610    
611                if (!result) {
612                    if (speex == null) {
613                        Toast.makeText(MainActivity.this,
614                                getResources().getString(R.string.prompt_file_speex_not_found), Toast.LENGTH_SHORT).show();
615                        return;
616                    }
617    
618                    Toast.makeText(MainActivity.this,
619                            getResources().getString(R.string.prompt_file_decoding_failed), Toast.LENGTH_SHORT).show();
620                    return;
621                }
622    
623                Toast.makeText(MainActivity.this,
624                        String.format(getResources().getString(R.string.prompt_file_decoded_successfully),
625                                speexTmp2.getFile().getAbsolutePath()),
626                        Toast.LENGTH_SHORT).show();
627            }
628        }

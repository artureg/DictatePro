001    package com.wiseapps.davacon.utils;
002    
003    import android.content.Context;
004    import android.graphics.Typeface;
005    
006    /**
007     * Helper class to work with fonts.
008     *
009     * @author varya.bzhezinskaya@gmail.com
010     *         Date: 3/20/14
011     *         Time: 12:26 PM
012     */
013    public class FontUtils {
014        private static Typeface robotoMedium;
015        private static Typeface robotoRegular;
016    
017        public static Typeface getRobotoMedium(Context context) {
018            if (robotoMedium == null) {
019                robotoMedium = Typeface.createFromAsset(context.getAssets(), "fonts/Roboto-Medium.ttf");
020            }
021    
022            return robotoMedium;
023        }
024    
025        public static Typeface getRobotoRegular(Context context) {
026            if (robotoRegular == null) {
027                robotoRegular = Typeface.createFromAsset(context.getAssets(), "fonts/Roboto-Regular.ttf");
028            }
029    
030            return robotoRegular;
031        }
032    }

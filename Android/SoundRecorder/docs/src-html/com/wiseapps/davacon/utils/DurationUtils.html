003    /**
004     * Helper class to provide the formatted durations.
005     *
006     * @author varya.bzhezinskaya@gmail.com
007     *         Date: 3/26/14
008     *         Time: 6:34 PM
009     */
010    public class DurationUtils {
011    
012        public static double format(int millis) {
013            return format((double) millis);
014        }
015    
016        public static double format(double millis) {
017            double seconds = millis / 1000;
018            return (double) Math.round(seconds * 10) / 10;
019        }
020    }

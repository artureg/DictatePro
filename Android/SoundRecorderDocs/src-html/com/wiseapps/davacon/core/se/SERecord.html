010    public class SERecord {
011        long start;
012        String soundPath;
013    
014        final SEProject project;
015    
016        SERecord prevRecord, nextRecord;
017    
018        // record current position in bytes
019        long position;
020    
021        // record duration in bytes
022        long duration;
023    
024        SERecord(SEProject project) {
025            this.project = project;
026        }
027    
028    //    SEAudioStream getAudioStream(Context context) {
029    //        return new SERecordAudioStream(this, context);
030    //    }
031    
032        AudioStream getAudioStream() {
033            return new RecordAudioStream(this);
034        }
035    
036        @Override
037        public boolean equals(Object o) {
038            if (this == o) return true;
039            if (o == null || getClass() != o.getClass()) return false;
040    
041            SERecord record = (SERecord) o;
042    
043            if (Double.compare(record.duration, duration) != 0) return false;
044            if (Double.compare(record.start, start) != 0) return false;
045            if (soundPath != null ? !soundPath.equals(record.soundPath) : record.soundPath != null) return false;
046    
047            return true;
048        }
049    
050        @Override
051        public int hashCode() {
052            int result;
053            long temp;
054            temp = Double.doubleToLongBits(start);
055            result = (int) (temp ^ (temp >>> 32));
056            temp = Double.doubleToLongBits(duration);
057            result = 31 * result + (int) (temp ^ (temp >>> 32));
058            result = 31 * result + (soundPath != null ? soundPath.hashCode() : 0);
059            return result;
060        }
061    }

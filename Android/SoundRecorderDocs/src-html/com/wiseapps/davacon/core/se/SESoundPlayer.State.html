001    package com.wiseapps.davacon.core.se;
002    
003    import android.media.AudioTrack;
004    import android.os.Handler;
005    
006    import java.util.ArrayList;
007    import java.util.List;
008    
009    import static com.wiseapps.davacon.core.se.SEProjectEngine.*;
010    
011    /**
012     * @author varya.bzhezinskaya@gmail.com
013     *         Date: 4/15/14
014     *         Time: 4:07 PM
015     */
016    class SESoundPlayer {
017        private static final String TAG = SESoundPlayer.class.getSimpleName();
018    
019        static final int MSG_PLAYING_STARTED = 0;
020        static final int MSG_PLAYING_IN_PROGRESS = 1;
021        static final int MSG_PLAYING_PAUSED = 2;
022        static final int MSG_PLAYING_STOPPED = 3;
023        static final int MSG_PLAYING_ERROR = 4;
024    
025        private enum State {
026            PLAYING, PAUSED, STOPPED
027        }
028        private State state;
029    
030        private final SEAudioStream stream;
031        private List<Handler> handlers = new ArrayList<Handler>();
032    
033        SESoundPlayer(SEAudioStream stream) {
034            this.stream = stream;
035        }
036    
037        void start() {
038            new Thread(new Runnable() {
039                @Override
040                public void run() {
041                    AudioTrack audioTrack = new AudioTrack(STREAM_TYPE, SAMPLE_RATE_IN_HZ, CHANNEL_CONFIG, AUDIO_FORMAT,
042                            MIN_BUFFER_SIZE, MODE);
043    
044                    if (audioTrack.getState() == AudioTrack.STATE_INITIALIZED) {
045                        sendMsgError();
046                        return;
047                    }
048    
049                    stream.open(SEAudioStream.Mode.READ);
050                    sendMsgStarted();
051    
052                    int offset = 0;
053                    byte data[] = stream.read(offset, 1);
054                    while(state != State.STOPPED) {
055                        if (state == State.PAUSED) {
056                            // TODO send just once
057                            sendMsgPaused();
058                        }
059    
060                        audioTrack.write(data, 0, data.length);
061                        audioTrack.play();
062    
063                        data = stream.read(offset, 1);
064    
065                        offset++;
066    
067                        sendMsgInProgress();
068                    }
069    
070                    stream.close();
071    
072                    audioTrack.stop();
073                    audioTrack.release();
074    
075                    sendMsgStopped();
076                }
077            }).start();
078    
079            state = State.PLAYING;
080        }
081    
082        void pause() {
083            state = State.PAUSED;
084        }
085    
086        void stop() {
087            state = State.STOPPED;
088        }
089    
090        void addHandler(Handler handler) {
091            handlers.add(handler);
092        }
093    
094        private void sendMsgStarted() {
095            for (Handler handler : handlers) {
096                handler.sendMessage(handler.obtainMessage(MSG_PLAYING_STARTED));
097            }
098        }
099    
100        private void sendMsgInProgress() {
101            for (Handler handler : handlers) {
102                handler.sendMessage(handler.obtainMessage(MSG_PLAYING_IN_PROGRESS));
103            }
104        }
105    
106        private void sendMsgPaused() {
107            for (Handler handler : handlers) {
108                handler.sendMessage(handler.obtainMessage(MSG_PLAYING_PAUSED));
109            }
110        }
111    
112        private void sendMsgStopped() {
113            for (Handler handler : handlers) {
114                handler.sendMessage(handler.obtainMessage(MSG_PLAYING_STOPPED));
115            }
116        }
117    
118        private void sendMsgError() {
119            for (Handler handler : handlers) {
120                handler.sendMessage(handler.obtainMessage(MSG_PLAYING_ERROR));
121            }
122        }
123    }

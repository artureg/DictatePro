001    package com.wiseapps.davacon.core.se;
002    
003    import android.media.AudioRecord;
004    import android.media.MediaRecorder;
005    import android.os.Handler;
006    
007    import java.util.ArrayList;
008    import java.util.List;
009    
010    import static com.wiseapps.davacon.core.se.SEProjectEngine.*;
011    
012    /**
013     * @author varya.bzhezinskaya@gmail.com

016     */
017    class SESoundRecorder {
018        private static final String TAG = SESoundRecorder.class.getSimpleName();
019    
020        static final int MSG_RECORDING_STARTED = 0;
021        static final int MSG_RECORDING_IN_PROGRESS = 1;
022        static final int MSG_RECORDING_STOPPED = 2;
023        static final int MSG_RECORDING_ERROR = 3;
024    
025        private boolean running;
026    
027        private final SEAudioStream stream;
028        private List<Handler> handlers = new ArrayList<Handler>();
029    
030        SESoundRecorder(SEAudioStream stream) {
031            this.stream = stream;
032        }
033    
034        void start() {
035            new Thread(new Runnable() {
036                @Override
037                public void run() {
038                    AudioRecord audioRecord = new AudioRecord(MediaRecorder.AudioSource.MIC,
039                            SAMPLE_RATE_IN_HZ, CHANNEL_CONFIG, AUDIO_FORMAT, MIN_BUFFER_SIZE);
040    
041                    if (audioRecord.getState() != AudioRecord.STATE_INITIALIZED) {
042                        sendMsgError();
043                        return;
044                    }
045    
046                    // TODO here we 1) check the streamRead current position and find the current record
047                    // TODO 2) split current record R on R1 and R2 (after a record is created, project .plist is auto updated)
048                    stream.open(SEAudioStream.Mode.WRITE);
049                    sendMsgStarted();
050    
051                    byte[] data = new byte[MIN_BUFFER_SIZE];
052                    while(running) {
053                        audioRecord.read(data, 0, data.length);
054                        stream.write(data);
055    
056                        sendMsgInProgress();
057                    }
058    
059                    stream.close();
060    
061                    audioRecord.stop();
062                    audioRecord.release();
063    
064                    sendMsgStopped();
065                }
066            }).start();
067    
068            running = true;
069        }
070    
071        void stop() {
072            running = false;
073        }
074    
075        void addHandler(Handler handler) {
076            handlers.add(handler);
077        }
078    
079        private void sendMsgStarted() {
080            for (Handler handler : handlers) {
081                handler.sendMessage(handler.obtainMessage(MSG_RECORDING_STARTED));
082            }
083        }
084    
085        private void sendMsgInProgress() {
086            for (Handler handler : handlers) {
087                handler.sendMessage(handler.obtainMessage(MSG_RECORDING_IN_PROGRESS));
088            }
089        }
090    
091        private void sendMsgStopped() {
092            for (Handler handler : handlers) {
093                handler.sendMessage(handler.obtainMessage(MSG_RECORDING_STOPPED));
094            }
095        }
096    
097        private void sendMsgError() {
098            for (Handler handler : handlers) {
099                handler.sendMessage(handler.obtainMessage(MSG_RECORDING_ERROR));
100            }
101        }
102    }

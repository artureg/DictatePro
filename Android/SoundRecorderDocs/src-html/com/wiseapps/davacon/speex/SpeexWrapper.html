001    package com.wiseapps.davacon.speex;
002    
003    /**
004     * Class provides native methods
005     *
006     * @author varya.bzhezinskaya@gmail.com
007     *         Date: 4/2/14
008     *         Time: 11:58 AM
009     */
010    public class SpeexWrapper {
011    
012        static {
013            System.loadLibrary("SpeexLib");
014        }
015    
016        /**
017         * Returns format of the file.
018         *
019         * @param filePath path to the file
020         * @return file format, 0 for standard wav, 1 for speex
021         */
022        native public static int getFormat(String filePath);
023    
024        /**
025         * Returns sample rate of the file.
026         *
027         * @param filePath path to the file
028         * @param format file format
029         * @return sample rate of the file or -1 if file doesn't exist
030         */
031        native public static int getSampleRate(String filePath, int format);
032    
033        /**
034         * Returns duration of the file.
035         *
036         * @param filePath path to the file
037         * @param format file format
038         * @return duration in seconds of the file or -1 if file doesn't exist
039         */
040        native public static double getDuration(String filePath, int format);
041    
042        /**
043         * Reads data from the file starting from offset to (offset + duration).
044         *
045         * @param filePath path to the file to read data from
046         * @param offset offset in seconds to start reading the data from
047         * @param duration duration in seconds of the data
048         * @param format file format
049         * @return data
050         */
051        native public static byte[] read(String filePath, double offset, double duration, int format);
052    
053        /**
054         * Writes data to the end of the file.
055         *
056         * @param filePath path to the file to write data from
057         * @param data data to be written
058         * @param format file format
059         * @return code of the operation, if success return 0
060         */
061        native public static int write(String filePath, byte[] data, int format);
062    
063        /**
064         * Encodes .wav file into RIFF/SPEEX
065         *
066         * @param wavFilePathStr name of the source .wav file
067         * @param compressedFilePathStr name of the dest RIFF/SPEEX file
068         * @return error code or 0 if success
069         */
070        native public static int encode(String wavFilePathStr, String compressedFilePathStr);
071    
072        /**
073         * Decodes RIFF/SPEEX file to .wav
074         *
075         * @param compressedFilePathStr name of the source RIFF/SPEEX file
076         * @param wavFilePathStr name of the dest .wav file
077         * @return error code or 0 if success
078         */
079        native public static int decode(String compressedFilePathStr, String wavFilePathStr);
080    
081        /**
082         * For testing purposes only, adds two digits.
083         *
084         * @param a the first digit to add
085         * @param b the second digit to add
086         * @return the sum
087         */
088        native public static int test(int a, int b);
089    }

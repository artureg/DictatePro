118        private void initWidgets() {
119            buttonRewind = (ImageButton) findViewById(R.id.rewind);
120            buttonRecord = (ImageButton) findViewById(R.id.record);
121            buttonForward = (ImageButton) findViewById(R.id.forward);
122    
123            buttonStart = (ImageButton) findViewById(R.id.start);
124            buttonPlay = (ImageButton) findViewById(R.id.play);
125            buttonEnd = (ImageButton) findViewById(R.id.end);

128            buttonSave = (ImageButton) findViewById(R.id.save);
129    
130            textDuration = (TextView) findViewById(R.id.duration);
131    
132            seekVolume = (SeekBar) findViewById(R.id.volume);
133            seekVolume.setMax(audioManager.getStreamMaxVolume(AudioManager.STREAM_SYSTEM));
134            seekVolume.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
135                @Override
136                public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
137                    audioManager.setStreamVolume(AudioManager.STREAM_SYSTEM, progress, AudioManager.FLAG_PLAY_SOUND);
138                }
139    
140                @Override
141                public void onStartTrackingTouch(SeekBar seekBar) {
142                }
143    
144                @Override
145                public void onStopTrackingTouch(SeekBar seekBar) {
146                }
147            });
148            seekVolume.setProgress(audioManager.getStreamVolume(AudioManager.STREAM_SYSTEM));
149    
150            seekPosition = (SeekBar) findViewById(R.id.position);
151            seekPosition.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
152                @Override
153                public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
154                    if (!fromUser) {
155                        return;
156                    }
157    
158                    if (engine.getState() != State.READY) {
159                        return;
160                    }
161    
162                    engine.setCurrentTime(DurationUtils.secondsToBytes((double) progress / 10));
163    
164                    textDuration.setText(
165                            String.format(getResources().getString(R.string.process_track_duration),
166                                    DurationUtils.secondsFromBytes(engine.getCurrentTime()),
167                                    DurationUtils.secondsFromBytes(engine.getDuration())));
168                }
169    
170                @Override
171                public void onStartTrackingTouch(SeekBar seekBar) {
172                }
173    
174                @Override
175                public void onStopTrackingTouch(SeekBar seekBar) {
176                }
177            });
178    
179            updateProgress();
180        }
181    
182        private void updateProgress() {
183            updateProgress((int) engine.getCurrentTime(), (int) engine.getDuration());
184        }
185    
186        private void updateProgress(int position, int duration) {
187            seekPosition.setProgress((int) (DurationUtils.secondsFromBytes(position) * 10));
188            seekPosition.setMax((int) (DurationUtils.secondsFromBytes(duration) * 10));
189    
190            textDuration.setText(
191                    String.format(getResources().getString(R.string.process_track_duration),
192                            DurationUtils.secondsFromBytes(position), DurationUtils.secondsFromBytes(duration)));
193        }
194    
195        public void rewind(View view) {
196            if (engine.getState() != State.READY) {
197                return;
198            }
199    
200            engine.setCurrentTime(engine.getCurrentTime() - DurationUtils.secondsToBytes(1));
201            updateProgress();
202        }
203    
204        public void record(View view) {
205            if (engine.getState() == State.READY) {
206                engine.startRecording();
207                return;
208            }
209    
210            if (engine.getState() == State.RECORDING_IN_PROGRESS) {
211                engine.stopRecording();
212            }
213        }
214    
215        public void forward(View view) {
216            if (engine.getState() != State.READY) {
217                return;
218            }
219    
220            engine.setCurrentTime(engine.getCurrentTime() + DurationUtils.secondsToBytes(1));
221            updateProgress();
222        }
223    
224        public void start(View view) {
225            if (engine.getState() != State.READY) {
226                return;
227            }
228    
229            engine.setCurrentTime(Long.MIN_VALUE);
230            updateProgress();
231        }
232    
233        public void play(View view) {
234            if (engine.getState() == State.READY) {
235                engine.startPlaying();
236                return;
237            }
238    
239            if (engine.getState() == State.PLAYING_IN_PROGRESS) {
240                engine.pausePlaying();
241            }
242        }
243    
244        public void end(View view) {
245            if (engine.getState() != State.READY) {
246                return;
247            }
248    
249            engine.setCurrentTime(Long.MAX_VALUE);
250            updateProgress();
251        }
252    
253        public void encode(View view) {
254            if (engine.getState() != State.READY) {
255                return;
256            }
257    
258            new EncodeProjectTask().execute();
259        }
260    
261        public void delete(View view) {
262            if (engine.getState() != State.READY) {
263                return;
264            }
265    
266            new DeleteProjectTask().execute();
267        }
268    
269        public void save(View view) {
270            if (engine.getState() != State.READY) {
271                return;
272            }
273    
274            new SaveProjectTask().execute();
275        }

508        private class SaveProjectTask extends AsyncTask<Void, Void, Boolean> {
509            @Override
510            protected void onPreExecute() {
511                super.onPreExecute();
512    
513                dialog = new ProgressDialog(getContext());
514    
515                dialog.show();
516    
517                dialog.setMessage(getResources().getString(R.string.project_save_in_progress));
518                dialog.setCancelable(false);
519            }
520    
521            @Override
522            protected Boolean doInBackground(Void... params) {
523                return project.save();
524            }
525    
526            @Override
527            protected void onPostExecute(Boolean aBoolean) {
528                super.onPostExecute(aBoolean);
529    
530                if (dialog != null) {
531                    dialog.dismiss();
532                    dialog = null;
533                }
534    
535                updateProgress();
536    
537                Toast.makeText(getContext(),
538                        aBoolean ? "Project saved successfully!" : "Saving project failed...",
539                        Toast.LENGTH_SHORT).show();
540            }
541        }
542    
543        private class DeleteProjectTask extends AsyncTask<Void, Void, Boolean> {
544            @Override
545            protected void onPreExecute() {
546                super.onPreExecute();
547    
548                dialog = new ProgressDialog(getContext());
549    
550                dialog.show();
551    
552                dialog.setMessage(getResources().getString(R.string.project_deletion_in_progress));
553                dialog.setCancelable(false);
554            }
555    
556            @Override
557            protected Boolean doInBackground(Void... params) {
558                boolean result = SDCardUtils.deleteProject(project);
559    
560                project = new SEProject(getContext());
561                SDCardUtils.readProject(project);
562    
563                return result;
564            }
565    
566            @Override
567            protected void onPostExecute(Boolean aBoolean) {
568                super.onPostExecute(aBoolean);
569    
570                if (dialog != null) {
571                    dialog.dismiss();
572                    dialog = null;
573                }
574    
575                destroyEngine();
576                initEngine();
577    
578                updateProgress();
579    
580                Toast.makeText(getContext(),
581                        aBoolean ? "Project deleted successfully!" : "Deletion of project failed...",
582                        Toast.LENGTH_SHORT).show();
583            }
584        }

204        // position and duration are relative units!
205        private void updatePositionSeekBar(long position, long duration) {
206            long secondInBytes = DurationUtils.secondsToBytes(1);
207            if ((this.duration + duration) / secondInBytes < 2 * 60) {
208                this.max = 2 * 60 * secondInBytes;
209    
210                playbackPos.setMax((int) this.max);
211                recordingPos.setMax((int) this.max);
212            } else {
213                if ((this.duration + duration) / secondInBytes < 10 * 60) {
214                    this.max = 10 * 60 * secondInBytes;
215    
216                    playbackPos.setMax((int) this.max);
217                    recordingPos.setMax((int) this.max);
218    
219                    // just in case the next time the above condition is false
220                    this.max = this.max * 2;
221                } else {
222                    if (++count * secondInBytes >= this.max * 2) {
223                    this.max = this.max * 2;
224                    }
225    
226                    playbackPos.setMax((int) (this.max));
227                    recordingPos.setMax((int) this.max);
228                }
229            }
230    
231            switch (engine.getState()) {
232                case READY: {
233                    recordingPos.setVisibility(View.GONE);
234    
235                    playbackPos.setThumb(
236                            getResources().getDrawable(R.drawable.play_red_cursor));
237                    playbackPos.setProgress((int) (this.position + position));
238                    playbackPos.setSecondaryProgress((int) (this.duration + duration));
239    
240                    break;
241                }
242                case PLAYING_IN_PROGRESS: {
243                    recordingPos.setVisibility(View.GONE);
244    
245                    playbackPos.setThumb(
246                            getResources().getDrawable(R.drawable.play_red_cursor));
247                    playbackPos.setProgress((int) (this.position + position));
248                    playbackPos.setSecondaryProgress((int) (this.duration + duration));
249    
250                    playbackPos.setThumb(getResources().getDrawable(R.drawable.play_red_cursor));
251    
252                    break;
253                }
254                case RECORDING_IN_PROGRESS: {
255    //                LoggerFactory.obtainLogger(TAG).
256    //                        d("updatePositionSeekBar# this.position = " + this.position + ", this.duration = " + this.duration +
257    //                                ", position = " + position + ", duration = " + duration + ", max = " + max + ", count = " + count);
258    
259                    if (recordingPos.getVisibility() == View.GONE) {
260                        recordingPos.setProgress((int) (this.position + position));
261                        recordingPos.setVisibility(View.VISIBLE);
262                    }
263    
264                    recordingPos.setSecondaryProgress((int) (this.position + position));
265    
266                    playbackPos.setThumb(null);
267                    playbackPos.setProgress((int) (this.position + position));
268                    playbackPos.setSecondaryProgress((int) (this.duration + duration));
269    
270                    break;
271                }
272            }
273        }

275        // position and duration are relative units!
276        private void updatePositionText(long position, long duration) {
277            double dP = DurationUtils.secondsFromBytes(this.position + position);
278            double dD = DurationUtils.secondsFromBytes(this.duration + duration);
279    
280            textDuration.setText(
281                    String.format(getResources().getString(R.string.process_track_duration),
282                            new DecimalFormat("0.0").format(dP), new DecimalFormat("0.0").format(dD)));
283        }

285        public void rewind(View view) {
286            if (engine.getState() != State.READY) {
287                return;
288            }
289    
290            engine.setCurrentTime(engine.getCurrentTime() - DurationUtils.secondsToBytes(1));
291    
292            position = engine.getCurrentTime();
293            duration = engine.getDuration();
294    
295            updatePositionSeekBar(0, 0);
296            updatePositionText(0, 0);
297        }
298    
299        public void record(View view) {
300            if (engine.getState() == State.READY) {
301                position = engine.getCurrentTime();
302                duration = engine.getDuration();
303    
304                engine.startRecording();
305                updateStateRecordingInProgress();
306                return;
307            }
308    
309            if (engine.getState() == State.RECORDING_IN_PROGRESS) {
310                engine.stopRecording();
311                updateStateReady();
312            }
313        }
314    
315        public void forward(View view) {
316            if (engine.getState() != State.READY) {
317                return;
318            }
319    
320            engine.setCurrentTime(engine.getCurrentTime() + DurationUtils.secondsToBytes(1));
321    
322            position = engine.getCurrentTime();
323            duration = engine.getDuration();
324    
325            updatePositionSeekBar(0, 0);
326            updatePositionText(0, 0);
327        }
328    
329        public void start(View view) {
330            if (engine.getState() != State.READY) {
331                return;
332            }
333    
334            engine.setCurrentTime(Long.MIN_VALUE);
335    
336            position = engine.getCurrentTime();
337            duration = engine.getDuration();
338    
339            updatePositionSeekBar(0, 0);
340            updatePositionText(0, 0);
341        }
342    
343        public void play(View view) {
344            if (engine.getState() == State.READY) {
345                position = engine.getCurrentTime();
346                duration = engine.getDuration();
347    
348                if (engine.getCurrentTime() >= engine.getDuration()) {
349                    engine.setCurrentTime(Long.MIN_VALUE);
350    
351                    position = engine.getCurrentTime();
352                    duration = engine.getDuration();
353    
354                    updatePositionSeekBar(0, 0);
355                    updatePositionText(0, 0);
356                }
357    
358                engine.startPlaying();
359                updateStatePlayingInProgress();
360                return;
361            }
362    
363            if (engine.getState() == State.PLAYING_IN_PROGRESS) {
364                engine.pausePlaying();
365                updateStateReady();
366            }
367        }
368    
369        public void end(View view) {
370            if (engine.getState() != State.READY) {
371                return;
372            }
373    
374            engine.setCurrentTime(Long.MAX_VALUE);
375    
376            position = engine.getCurrentTime();
377            duration = engine.getDuration();
378    
379            updatePositionSeekBar(0, 0);
380            updatePositionText(0, 0);
381        }

502            @Override
503            public void playingPaused(final int position, final int duration) {
504                runOnUiThread(new Runnable() {
505                    @Override
506                    public void run() {
507                        updatePositionSeekBar(position, duration);
508                        updatePositionText(position, duration);

579        private class SaveProjectTask extends AsyncTask<Void, Void, Boolean> {
580            @Override
581            protected void onPreExecute() {
582                super.onPreExecute();
583    
584                dialog = new ProgressDialog(getContext());
585    
586                dialog.show();
587    
588                dialog.setMessage(getResources().getString(R.string.project_save_in_progress));
589                dialog.setCancelable(false);
590            }
591    
592            @Override
593            protected Boolean doInBackground(Void... params) {
594                return project.save();
595            }
596    
597            @Override
598            protected void onPostExecute(Boolean aBoolean) {
599                super.onPostExecute(aBoolean);
600    
601                if (dialog != null) {
602                    dialog.dismiss();
603                    dialog = null;
604                }
605    
606                position = engine.getCurrentTime();
607                duration = engine.getDuration();
608    
609                updatePositionSeekBar(0, 0);
610                updatePositionText(0, 0);
611    
612                Toast.makeText(getContext(),
613                        aBoolean ? "Project saved successfully!" : "Saving project failed...",
614                        Toast.LENGTH_SHORT).show();
615            }
616        }

618        private class DeleteProjectTask extends AsyncTask<Void, Void, Boolean> {
619            @Override
620            protected void onPreExecute() {
621                super.onPreExecute();
622    
623                dialog = new ProgressDialog(getContext());
624    
625                dialog.show();
626    
627                dialog.setMessage(getResources().getString(R.string.project_deletion_in_progress));
628                dialog.setCancelable(false);
629            }
630    
631            @Override
632            protected Boolean doInBackground(Void... params) {
633                boolean result = SDCardUtils.deleteProject(project);
634    
635                project = new SEProject(getContext());
636                SDCardUtils.readProject(project);
637    
638                return result;
639            }
640    
641            @Override
642            protected void onPostExecute(Boolean aBoolean) {
643                super.onPostExecute(aBoolean);
644    
645                if (dialog != null) {
646                    dialog.dismiss();
647                    dialog = null;
648                }
649    
650                destroyEngine();
651                initEngine();
652    
653                position = engine.getCurrentTime();
654                duration = engine.getDuration();
655    
656                updatePositionSeekBar(0, 0);
657                updatePositionText(0, 0);
658    
659                Toast.makeText(getContext(),
660                        aBoolean ? "Project deleted successfully!" : "Deletion of project failed...",
661                        Toast.LENGTH_SHORT).show();
662            }
663        }

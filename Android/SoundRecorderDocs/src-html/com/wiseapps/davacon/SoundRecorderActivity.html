061        private void initData() {
062            try {
063                File root = FileUtils.getRoot(this);
064    
065                String[] filenames = root.list(new FilenameFilter() {
066                    @Override
067                    public boolean accept(File dir, String filename) {
068                        return filename.endsWith(".plist");
069                    }
070                });
071    
072                if (filenames != null && filenames.length != 0) {
073                    project = new SEProject(new File(filenames[0]).getAbsolutePath());
074                    return;
075                }
076    
077                project = new SEProject(FileUtils.createProjectPath(getContext()));
078            } catch (Exception e) {
079                LoggerFactory.obtainLogger(TAG).
080                        e("initData# ", e);
081            }
082    
083            if (project != null) {
084                engine = new SEProjectEngine(getContext(), project);
085                engine.addPlayerStateListener(playerStateListener);
086                engine.addRecorderStateListener(recorderStateListener);
087            }
088        }
089    
090        private void initWidgets() {
091            buttonRecord = (ImageButton) findViewById(R.id.record);
092    
093            volumeBar = (SeekBar) findViewById(R.id.volume);
094            volumeBar.setMax(audioManager.getStreamMaxVolume(AudioManager.STREAM_SYSTEM));
095            volumeBar.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
096                @Override
097                public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
098                    audioManager.setStreamVolume(AudioManager.STREAM_SYSTEM, progress, AudioManager.FLAG_PLAY_SOUND);
099                }

101                @Override
102                public void onStartTrackingTouch(SeekBar seekBar) {
103                }
104    
105                @Override

107                }
108            });
109        }
